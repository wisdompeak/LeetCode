/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        if (head == NULL || head->next == NULL)
         return NULL;
     
      ListNode *x  = head;
      ListNode *y  = head;
      ListNode *z = head;
     
      while (y->next && y->next->next) {
          x = x->next;
          y = y->next->next;
          if (x == y) {
              while(x != z) {               
                 x  = x->next;
                 z = z->next;
             }
             return z;
         }
      }
     return NULL;                          
    }
};
